# 题解
```js
/**
 * 双指针
 * @param {number[]} numbers
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(numbers, target) {
    let i = 0;
    let j = numbers.length - 1;
    while(i < j){
        let val = numbers[i] + numbers[j];
        if(val === target){
            return [ ++i, ++j];
        }else if(val > target){
            j--
        }else{
            i++;
        }
    }
    return [-1,-1]
};
```
