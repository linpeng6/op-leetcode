# 题目地址
https://leetcode-cn.com/problems/third-maximum-number/

# 题解
```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var thirdMax = function (nums) {
    let set = new Set(nums);
    let arr = [...set];
    arr.sort((a,b)=> a - b);
    return arr.length >= 3 ? arr[arr.length - 3] : arr[arr.length - 1]
};
```
